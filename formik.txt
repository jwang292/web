import * as Yup from 'yup'
validationSchema: Yup.object({
      mobile: Yup.string()
        .required('phone cannot be empty')
        .matches(/^1\d{10}$/, 'phone formate is wrong'),
      code: Yup.string()
        .required('code cannot be empty')
        .matches(/^\d{6}$/, 'phone formate is wrong'),
    })
